keycodes:





in grafic-window:
  Ctrl-S      Save model
  Ctrl-Z      delete last created object
  Ctrl-Y      restore last deleted object
  Ctrl-P      redo last user-application


in editor-window:
  Ctrl-S      Save model
  Shift-cursorKey  select text
  Ctrl-C      copy selected text
  Ctrl-X      delete & copy selected text
  Ctrl-V      paste previous copied text
  Esc         select object-ID/expression left of cursor
  Ctrl-P      redo last user-application


in browser-window:
  Ctrl-P      redo last user-application
